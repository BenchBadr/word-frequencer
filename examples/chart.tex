\usepackage{tikz}

\newcommand{\chart}[4]{
\begin{tikzpicture}[smooth,scale=1]
    
  
    % reglages 
    \def\taille{5} % taille de la grille en cm


    % Grille
    \draw[very thin,color=gray!30, scale=1] (0,0) grid (\taille,\taille);

    % Axes
    \draw[->] (0,0) -- (\taille+0.2,0) node[right]{#3};
    \draw[->] (0,0) -- (0,\taille+0.2) node[above]{#4};

    % graduations
    \foreach \n in {0,...,5}{
        \pgfmathsetmacro\N{int(\n * #1/\taille)}
        \node[below] at (\n, 0) {$\N$};
        \ifnum\n=0
        \else
            \node[anchor=center] at (-.4, \n) {$\N$};
      \fi
    }

    % Courbe
    \begin{scope}
      \clip (0,0) rectangle (\taille,\taille);
    \draw[color=blue,thick, scale=\taille/#1] plot[smooth,tension=0.5] coordinates {#2};
    \end{scope}
\end{tikzpicture}}
